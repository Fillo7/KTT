544  {
545  if (!std::is_trivially_copyable<T>() || typeid(T) == typeid(bool))
546  {
547  std::cerr << "Unsupported argument data type" << std::endl;
548  throw std::runtime_error("Unsupported argument data type");
549  }
550 
551  if (!std::is_arithmetic<T>())
552  {
553  return ArgumentDataType::Custom;
554  }
555 
556  if (sizeof(T) == 1 && std::is_unsigned<T>())
557  {
558  return ArgumentDataType::UnsignedChar;

562  return ArgumentDataType::Char;

564  else if (typeid(T) == typeid(half))
