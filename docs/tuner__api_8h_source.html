646  template <typename T> ArgumentDataType getMatchingArgumentDataType() const
647  {
648  if (!std::is_trivially_copyable<T>() || typeid(T) == typeid(bool))
649  {
650  std::cerr << "Unsupported argument data type" << std::endl;
651  throw std::runtime_error("Unsupported argument data type");
652  }
653 
654  if (!std::is_arithmetic<T>())
655  {
656  return ArgumentDataType::Custom;
657  }
658 
659  if (sizeof(T) == 1 && std::is_unsigned<T>())
660  {
661  return ArgumentDataType::UnsignedChar;
662  }
663  else if (sizeof(T) == 1)
664  {
665  return ArgumentDataType::Char;
666  }
667  else if (typeid(T) == typeid(half))
668  {
669  return ArgumentDataType::Half;
670  }
671  else if (sizeof(T) == 2 && std::is_unsigned<T>())
672  {
673  return ArgumentDataType::UnsignedShort;
674  }
675  else if (sizeof(T) == 2)
676  {
677  return ArgumentDataType::Short;
678  }
679  else if (typeid(T) == typeid(float))
680  {
681  return ArgumentDataType::Float;
682  }
683  else if (sizeof(T) == 4 && std::is_unsigned<T>())
684  {
685  return ArgumentDataType::UnsignedInt;
686  }
687  else if (sizeof(T) == 4)
688  {
689  return ArgumentDataType::Int;
