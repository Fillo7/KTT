737  template <typename T> ArgumentDataType getMatchingArgumentDataType() const
738  {
739  static_assert(std::is_trivially_copyable<T>() && !std::is_reference<T>() && !std::is_pointer<T>(), "Unsupported argument data type");
740  static_assert(!std::is_same<std::remove_cv_t<T>, bool>(), "Bool argument data type is not supported");
741 
742  if (std::is_same<std::remove_cv_t<T>, half>())
743  {
744  return ArgumentDataType::Half;
745  }
746 
747  if (!std::is_arithmetic<T>())
748  {
749  return ArgumentDataType::Custom;
750  }
751 
752  if (sizeof(T) == 1 && std::is_unsigned<T>())
753  {
754  return ArgumentDataType::UnsignedChar;
755  }
756  else if (sizeof(T) == 1)
757  {
758  return ArgumentDataType::Char;
759  }
760  else if (sizeof(T) == 2 && std::is_unsigned<T>())
761  {
762  return ArgumentDataType::UnsignedShort;
763  }
764  else if (sizeof(T) == 2)
765  {
766  return ArgumentDataType::Short;
767  }
768  else if (std::is_same<std::remove_cv_t<T>, float>())
769  {
770  return ArgumentDataType::Float;
771  }
772  else if (sizeof(T) == 4 && std::is_unsigned<T>())
773  {
774  return ArgumentDataType::UnsignedInt;
775  }
776  else if (sizeof(T) == 4)
777  {
778  return ArgumentDataType::Int;
779  }
780  else if (std::is_same<std::remove_cv_t<T>, double>())
781  {
782  return ArgumentDataType::Double;
783  }
784  else if (sizeof(T) == 8 && std::is_unsigned<T>())
785  {
786  return ArgumentDataType::UnsignedLong;
787  }
788  else if (sizeof(T) == 8)
789  {
790  return ArgumentDataType::Long;
